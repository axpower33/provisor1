
&НаСервере
функция ПечатьДокументаНаСервере()
	ТабДокумент = Новый ТабличныйДокумент;
	ТабДокумент.Очистить();
	ДокОбъект=ЭтотОбъект.Объект;
	
	Макет=Документы.Приходноменклатуры.ПолучитьМакет("Макет");
	Шапка=Макет.ПолучитьОбласть("Шапка");
	Шапка.Параметры.Номер=ДокОбъект.Номер;
	Шапка.Параметры.ДатаДок=ДокОбъект.Дата;
	Шапка.Параметры.Контрагент=ДокОбъект.Контрагент.Наименование;
	
	ТабДокумент.Вывести(Шапка);
	
	Строка=Макет.ПолучитьОбласть("Строка");
	
	Для каждого Товар Из ДокОбъект.Товары Цикл
		Строка.Параметры.Номенклатура=Товар.Номенклатура;
		Строка.Параметры.ЕдИзм=Товар.Номенклатура.ЕдИзм;
		Строка.Параметры.Количество=Товар.Количество;
		
		Строка.Параметры.Цена=Товар.Цена;
		
		Строка.Параметры.Сумма=Товар.Сумма;
		
		ТабДокумент.Вывести(Строка);
	КонецЦикла;
	ТабДокумент.Показать();
	
	ТабДокумент.ЗакончитьАвтоГруппировкуСтрок();
	ТабДокумент.ВывестиГоризонтальныйРазделительСтраниц();
	
	
	ТабДокумент.ТолькоПросмотр=Истина;
	Возврат ТабДокумент;

КонецФункции


&НаКлиенте
Процедура ПечатьДокумента(Команда1)
    ТабДок = ПечатьДокументаНаСервере();
КонецПроцедуры
